#include<iostream>
#include<fstream>
using namespace std;
int main() {
    fstream f("textfilein.txt",ios::in);
    string s;
    int lines=0,digits=0;
    while (getline(f,s)) {
        cout<<s<<endl;
        for (int i=0;i<s.length();++i)
            if (s[i]>='0' and s[i]<='9')
                digits++;
        lines++;
    }
    cout<<lines<<endl;
    cout<<digits<<endl;
    f.close();
}
